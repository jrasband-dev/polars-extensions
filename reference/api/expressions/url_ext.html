Examples

import polars as pl
import polars_extensions as plx

df = pl.DataFrame({
    "link": [
        "https://pypi.org/search/?q=polars-extensions",
        "https://pypi.org/search/?q=polars",
        "https://pypi.org/search/?q=pyodbc"
    ]
})
df.with_columns(pl.col('link').url_ext.query().alias('q'))
