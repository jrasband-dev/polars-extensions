Convert Fahrenheit to Celsius.

Examples

import polars as pl
import polars_extensions as plx

df = pl.DataFrame({
    "temp_F": [32, 68, 104],
})
df.with_columns([
    pl.col("temp_F").unit_ext.fahrenheit_to_celsius().alias("temp_C"),
])

shape: (3, 2)
┌────────┬────────┐
│ temp_F ┆ temp_C │
│ ---    ┆ ---    │
│ i64    ┆ f64    │
╞════════╪════════╡
│ 32     ┆ 0.0    │
│ 68     ┆ 20.0   │
│ 104    ┆ 40.0   │
└────────┴────────┘

yards_to_meters() → Expr[source]#

Examples

import polars as pl
import polars_extensions as plx

df = pl.DataFrame({
    "yards": [10, 50, 100]
})

df.with_columns([
    pl.col("yards").unit_ext.yards_to_meters().alias("meters")
])

shape: (3, 2)
┌───────┬────────┐
│ yards ┆ meters │
│ ---   ┆ ---    │
│ i64   ┆ f64    │
╞═══════╪════════╡
│ 10    ┆ 9.144  │
│ 50    ┆ 45.72  │
│ 100   ┆ 91.44  │
└───────┴────────┘
